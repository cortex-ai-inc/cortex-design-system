Sheets are side panels powered by @radix-ui/react-dialog. Available sides: top, right (default), bottom, left. The overlay uses bg-black/50. Width defaults to w-3/4 sm:max-w-sm. Drawers use vaul for mobile-first bottom sheets with drag-to-dismiss behavior.

API

Prop Type Default Description
open boolean false Controlled open state.
onOpenChange (open: boolean) => void -- Callback when open state changes.
side "top" | "right" | "bottom" | "left" "right" Side from which the sheet slides in.

Usage

{`import {
  Sheet,
  SheetTrigger,
  SheetContent,
  SheetHeader,
  SheetTitle,
  SheetDescription,
} from '@/components/ui/sheet'

// Sheet (side panel)
<Sheet>
  <SheetTrigger asChild>
    <Button>Open Sheet</Button>
  </SheetTrigger>
  <SheetContent side="right">
    <SheetHeader>
      <SheetTitle>Ticket Details</SheetTitle>
      <SheetDescription>
        View and edit ticket information.
      </SheetDescription>
    </SheetHeader>
    {/* Content */}
  </SheetContent>
</Sheet>

// Drawer (bottom sheet - vaul)
import { Drawer } from 'vaul'

<Drawer.Root>
  <Drawer.Trigger asChild>
    <Button>Open Drawer</Button>
  </Drawer.Trigger>
  <Drawer.Portal>
    <Drawer.Overlay className="fixed inset-0 bg-black/50" />
    <Drawer.Content className="...">
      <div className="mx-auto w-12 h-1.5 rounded-full bg-muted" />
      <Drawer.Title>Title</Drawer.Title>
      <Drawer.Description>Description</Drawer.Description>
    </Drawer.Content>
  </Drawer.Portal>
</Drawer.Root>`}
